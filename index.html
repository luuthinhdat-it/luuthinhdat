async function traCuuDiem() { const sbd = document.getElementById("searchKey").value.trim(); const resultBox = document.getElementById("result"); const errorBox = document.getElementById("error"); if (!sbd) { alert("Vui lòng nhập Số báo danh!"); return; } // Ẩn bảng kết quả cũ đi trước khi tìm kiếm mới resultBox.style.display = "none"; errorBox.style.display = "none"; try { // GỌI ĐẾN FILE API BẠN VỪA TẠO const response = await fetch(`/api/tracuu?sbd=${sbd}`); if (!response.ok) { throw new Error("Lỗi kết nối"); } const data = await response.json(); // Nếu có dữ liệu trả về từ Sở và không có lỗi if (data && !data.error) { errorBox.style.display = "none"; resultBox.style.display = "block"; // Đổ dữ liệu thật từ Sở ra màn hình điện thoại // Lưu ý: Tên các biến (data.hoTen, data.toan...) phải khớp với API của Sở document.getElementById("resTen").innerText = data.hoTen || data.HoTen || "Thí sinh"; document.getElementById("resSBD").innerText = data.soBaoDanh || data.sbd || sbd; document.getElementById("resMaDD").innerText = data.maDinhDanh || "Đang cập nhật"; document.getElementById("resTruong").innerText = data.truongCu || "Đang cập nhật"; document.getElementById("resHoiDong").innerText = data.hoiDongThi || "Đang cập nhật"; document.getElementById("resToan").innerText = data.diemToan || data.toan || "0"; document.getElementById("resVan").innerText = data.diemVan || data.van || "0"; document.getElementById("resAnh").innerText = data.diemAnh || data.anh || "0"; document.getElementById("resChuyen").innerText = data.diemChuyen || "Không thi chuyên"; } else { resultBox.style.display = "none"; errorBox.style.display = "block"; // Hiện lỗi màu đỏ } } catch (err) { resultBox.style.display = "none"; errorBox.style.display = "block"; // Hiện lỗi màu đỏ nếu sập mạng } }
